\subparagraph{A.1.1}
Найдите простое выражение для $\sum_{k=1}^n {2k-1}$.
\begin{gather*}
\sum_{k=1}^n {2k-1} = 2\sum_{k=1}^n {k} - \sum_{k=1}^n {1} = n(n-1) - n = n^2.
\end{gather*}

\subparagraph{A.1.2}
Покажите, используя формулу для гармонического ряда, что $\sum_{k=1}^n {1/(2k-1)} = ln{\sqrt{n}} + O(1)$.
\begin{gather*}
H_{2n} = \sum_{k=1}^n {\frac{1}{(2k-1)}} + \sum_{k=1}^{n} {\frac{1}{2k}} \\
\sum_{k=1}^{n} {\frac{1}{k}} = H_n \\ 
\sum_{k=1}^{n} {\frac{1}{2k-1}} = H_{2n} - \frac{1}{2}H_n \\
H_n = \ln{n} + O(1)
\sum_{k=1}^{n} {\frac{1}{2k-1}} = \ln{2n} + O(1) - 1/2(\ln{n} + O(1)) = \\
\ln{2} + \ln{n} - 1/2(\ln{n}) + O(1) + 1/2O(1) = \ln{n} - 1/2(\ln{n}) + O(1) = \\ 
\ln{n}*(1 - 1/2) + O(1) = \frac{1}{2}\ln{n} + O(1) = \ln{\sqrt{n}} + O(1)
\end{gather*}

\subparagraph{A.1.3}
Покажите, что для $\left\lvert{x}\right\rvert < 1$ справедливо сотношение $\sum_{k=0}^{\infty} {k^2x^k} = x(1+x)/(1-x)^3$.
\begin{gather*}
\left\lvert{x}\right\rvert < 1 \\
\left(\sum_{k=0}^{\infty} {x^k}\right)' * x = \left( \frac{1}{1-x}\right)' * x \\
\left(\sum_{k=0}^{\infty} {kx^k}\right)' * x = \left( \frac{x}{(1-x)^2}\right)' * x \\
\sum_{k=0}^{\infty} {k^2x^k} = \left( \frac{(1-x)^2 - 2x(1 - x)(-1)}{(1-x)^4}\right) * x \\
= \left( \frac{(1-x)^2 + 2x(1 - x)}{(1-x)^4}\right) * x = \\
\left( \frac{(x+1)}{(1-x)^3}\right) * x = \frac{x(x+1)}{(1-x)^3}
\end{gather*}

\subparagraph{A.1.7}
Вычислите произведение $\prod_{k=1}^{n} {2*4^k}$.
\begin{gather*}
\lg{\left(\prod_{k=1}^{n} {2*4^k}\right)} = \sum_{k=1}^{n} {\lg{2*4^k}} \\
\sum_{k=1}^{n} { \left(\lg{2} + k\lg{4}\right) } = n\lg{2} + \lg{4} * \sum_{k=1}^{n} { \left(k\right) } = \\
n\lg{2} + \lg{4}\frac{n(n+1)}{2} = \\
n\lg{2} + \lg{2}n(n+1) = \lg{2}n(n+2) = (n^2 + 2n)\lg{2}
\end{gather*}

\subparagraph{A.1.8}
Вычислите произведение $\prod_{k=2}^{n} {(1 - 1/k^2)}$.
\begin{gather*}
\prod_{k=2}^{n} {\frac{k^2 - 1}{k^2}} = \\
\prod_{k=2}^{n} {\frac{(k - 1) * (k + 1)}{k^2}} = \\
\prod_{k=2}^{n} {\frac{k - 1}{k}} * \prod_{k=2}^{n} {\frac{k + 1}{k}} \\
\prod_{k=2}^{n} {\frac{k - 1}{k}} = \frac{1}{2} * \frac{2}{3} * \frac{3}{4} * \, \dots * \, \frac{n-1}{n} = \frac{1}{n} \\ 
\prod_{k=2}^{n} {\frac{k + 1}{k}} = \frac{3}{2} * \frac{4}{3} * \frac{5}{4} * \, \dots * \, \frac{n+1}{n} = \frac{n+1}{2} \\
\prod_{k=2}^{n} {(1 - 1/k^2)} = \frac{1}{n} * \frac{n+1}{2} = \frac{n+1}{2n}
\end{gather*}
